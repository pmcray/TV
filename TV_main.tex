\def\_printchap#1{%
   \vfill\eject
   \ifodd\pageno \else \blankpage \fi 
   \blankpage 
   \null\vskip 50mm
   {\leftskip=0pt plus 1fill \rightskip=0pt plus 1fill \parindent=0pt
    {\Albertus\typosize[26/30] #1} \par}
   \vskip 25mm
}

\def\_printsec#1{%
   \vfill\eject
   \ifodd\pageno \blankpage \fi 
   \begingroup
      \null\vskip 12mm
      {\leftskip=0pt plus 1fill \rightskip=0pt plus 1fill \parindent=0pt
       #1 \par}
      \vskip 12mm
   \endgroup
   \suppressindent
}
