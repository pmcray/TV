\def\_printchap#1{%
   \partpagetrue
   \ifnum\partnum=1
      % Already handled by front matter alignment
   \else
      \vfill\eject
      \blankpage % Move from Verso to next Recto
   \fi
   \null\vskip 12mm
   % Header and Rule - Explicitly left justified
   {\parindent=0pt \leftskip=0pt \rightskip=0pt \hsize=116mm
    \noindent {\serifboldlarge PART \the\partnum.\quad #1} \par 
    \vskip 2mm \hrule height 0.4pt \vskip 15mm}
   \immediate\write 1{\string\_Xpage{\lastphysicalpage}{\getpagenum}}%
   \immediate\write 1{\string\_Xtoc{1}{chap}{\the\partnum}{}{#1}}%
}
\def\_printsec#1{%
   \vfill\eject
   \partpagefalse
   \ifodd\pageno \blankpage \fi % Ensure we are on a Verso page
   \null\vskip 12mm
   {\parindent=0pt \leftskip=0pt \rightskip=0pt \hsize=116mm
    {\serifbold\typosize[14/16] #1} \par \vskip 2mm \hrule height 0.4pt \vskip 4mm}
   \suppressindent
}

% CUSTOM MACROS THAT CALL STANDARD OpTeX \chap AND \sec
\def\parttitle #1#2#3#4#5#6#7{%
   \xdef\currparttitle{#1}%
   \xdef\currmonth{}% 
   \xdef\currdaytitle{}%
   \advance\partnum by 1
   % If this is Part 7, we shift pageno by 2 BEFORE printing the chapter page
   \ifnum\partnum=7 
      \immediate\write 1{\string\_Xpage{\lastphysicalpage}{376}}%
      \immediate\write 1{\string\_Xtoc{2}{sec}{6.31}{}{It Doesn't Look Like Anywhere}}%
      \global\advance\pageno by 2 
   \fi
   \_printchap{#1}%
   % Epigraphs follow on same page
   \bookepigraph{#2}{#3}{#4}
   \bookepigraph{#5}{#6}{#7}
   \vfill\eject
   \partpagefalse
}
